            # 安装及使用

            nvmopen in new window 是 node 的版本管理工具，可以操作 node 版本的切换、安装.......等操作

            # 安装

            # 第一步：下载安装

            使用 curl 安装，官方不推荐使用 homebrew 安装。

            curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.37.2/install.sh | bash

            此处 v0.37.2 为 nvm 版本号，可以使用其他版本

            # 第二步：配置环境

            mac 下编辑 ~/.bash_profile 文件

            export NVM_DIR="$HOME/.nvm"
[ -s "$NVM_DIR/nvm.sh" ] && . "$NVM_DIR/nvm.sh"

# 可配置国内镜像
# export NVM_NODEJS_ORG_MIRROR=http://npm.taobao.org/mirrors/node
# export NVM_IOJS_ORG_MIRROR=http://npm.taobao.org/mirrors/iojs

            # 第三步：重新引用配置文件

            source ~/.bash_profile
source ~/.zshrc

            # 测试是否安装配置成功

            nvm -v 
# 0.38.0

            # 常用命令

            命令解析
            nvm install [version]安装指定版本
            nvm install stable安装最新版本
            nvm uninstall [version]卸载指定版本
            nvm use [version]切换指定版本
            nvm current查看当前版本
            nvm ls查看版本列表
            nvm ls [version]查看指定版本
            nvm ls-remote查看远程所有可用版本号
            nvm ls-remote [version]查看远程所有指定版本
            nvm alias [name] [version]给某个版本设置别名
            nvm alias default [version]指定某个版本为默认版本
            nvm unalias [name]删除别名
            nvm which显示安装路径
            nvm cache dir显示 nvm 的缓存目录
            nvm cache clear清除缓存
